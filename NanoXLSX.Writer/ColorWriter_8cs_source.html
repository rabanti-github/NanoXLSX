29 public IEnumerable<XmlAttribute> GetAttributes(Color color)
30 {
31 List<XmlAttribute> attributes = new List<XmlAttribute>();
32 if (color == null)
33 {
34 return attributes;
35 }
36 string name = GetAttributeName(color);
37 if (name != null)
38 {
39 string value = GetAttributeValue(color);
40 attributes.Add(new XmlAttribute(name, value));
41 }
42 if (UseTintAttribute(color))
43 {
44 attributes.Add(new XmlAttribute("tint", GetTintAttributeValue(color)));
45 }
46 return attributes;
47 }
48
54 public string GetAttributeName(Color color)
55 {
56 if (color == null)
57 {
58 return null;
59 }
60 switch (color.Type)
61 {
62 case Color.ColorType.Auto:
63 return "auto";
64 case Color.ColorType.Rgb:
65 return "rgb";
66 case Color.ColorType.Indexed:
67 return "indexed";
68 case Color.ColorType.Theme:
69 return "theme";
70 case Color.ColorType.System:
71 return "system";
72 default:
73 return null;
74 }
75 }
76
82 public string GetAttributeValue(Color color)
83 {
84 if (color == null)
85 {
86 return null;
87 }
88 switch (color.Type)
89 {
90 case Color.ColorType.Auto:
91 return "1";
92 case Color.ColorType.Rgb:
93 return color.RgbColor.StringValue;
94 case Color.ColorType.Indexed:
95 return color.IndexedColor.StringValue;
96 case Color.ColorType.Theme:
97 return color.ThemeColor.StringValue;
98 case Color.ColorType.System:
99 return color.SystemColor.StringValue;
100 default:
101 return null;
102 }
103 }
104
110 public string GetTintAttributeValue(Color color)
111 {
112 if (!color.Tint.HasValue)
113 {
114 return null;
115 }
116 double tint = color.Tint.Value;
117 return tint == 0.0 ? null : ParserUtils.ToString(tint);
118 }
119
125 public bool UseTintAttribute(Color color)
126 {
127 return color != null && color.Type == Color.ColorType.Theme && (!color.Tint.HasValue || !Comparators.IsZero(color.Tint.Value));
128 }
129 }
