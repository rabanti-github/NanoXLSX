59 public void Execute()
60 {
61 Theme workbookTheme = Workbook.WorkbookTheme;
62 theme = XmlElement.CreateElement("theme", "a");
63 theme.AddNameSpaceAttribute("a", "xmlns", "http://schemas.openxmlformats.org/drawingml/2006/main");
64 theme.AddAttribute("name", XmlUtils.SanitizeXmlValue(workbookTheme.Name));
65 XmlElement themeElements = theme.AddChildElement("themeElements", "a");
66 themeElements.AddChildElement(GetColorSchemeElement(workbookTheme.Colors));
67
68 WriterPlugInHandler.HandleInlineQueuePlugins(ref themeElements, Workbook, PlugInUUID.ThemeInlineWriter);
69 }

76 private XmlElement GetColorSchemeElement(ColorScheme scheme)
77 {
78 XmlElement colorScheme = XmlElement.CreateElementWithAttribute("clrScheme", "name", XmlUtils.SanitizeXmlValue(scheme.Name), "a");
79 colorScheme.AddChildElement(GetColor("dk1", scheme.Dark1, "a"));
80 colorScheme.AddChildElement(GetColor("lt1", scheme.Light1, "a"));
81 colorScheme.AddChildElement(GetColor("dk2", scheme.Dark2, "a"));
82 colorScheme.AddChildElement(GetColor("lt2", scheme.Light2, "a"));
83 colorScheme.AddChildElement(GetColor("accent1", scheme.Accent1, "a"));
84 colorScheme.AddChildElement(GetColor("accent2", scheme.Accent2, "a"));
85 colorScheme.AddChildElement(GetColor("accent3", scheme.Accent3, "a"));
86 colorScheme.AddChildElement(GetColor("accent4", scheme.Accent4, "a"));
87 colorScheme.AddChildElement(GetColor("accent5", scheme.Accent5, "a"));
88 colorScheme.AddChildElement(GetColor("accent6", scheme.Accent6, "a"));
89 colorScheme.AddChildElement(GetColor("hlink", scheme.Hyperlink, "a"));
90 colorScheme.AddChildElement(GetColor("folHlink", scheme.FollowedHyperlink, "a"));
91 return colorScheme;

101 private XmlElement GetColor(string name, IColor color, string prefix)
102 {
103 XmlElement colorElement = XmlElement.CreateElement(name, prefix);
104 if (color is SystemColor)
105 {
106 SystemColor sysColor = color as SystemColor;
107 XmlElement sysColorElement = colorElement.AddChildElementWithAttribute("sysClr", "val", sysColor.StringValue, "a");
108 if (!string.IsNullOrEmpty(sysColor.LastColor))
109 {
110 sysColorElement.AddAttribute("lastClr", sysColor.LastColor);
111 }
112 }
113 else if (color is SrgbColor)
114 {
115 colorElement.AddChildElementWithAttribute("srgbClr", "val", color.StringValue, "a");
116 }
117 return colorElement;
118 }
