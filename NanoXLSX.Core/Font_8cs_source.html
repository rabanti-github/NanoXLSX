402 private void ValidateFontScheme()
403 {
404 if ((string.IsNullOrEmpty(name)) && !StyleRepository.Instance.ImportInProgress)
405 {
406 throw new StyleException("The font name was null or empty");
407 }
408 if (name.Equals(DefaultMinorFont, System.StringComparison.Ordinal))
409 {
410 Scheme = SchemeValue.Minor;
411 }
412 else if (name.Equals(DefaultMajorFont, System.StringComparison.Ordinal))
413 {
414 Scheme = SchemeValue.Major;
415 }
416 else
417 {
418 Scheme = SchemeValue.None;
419 }
420 }
