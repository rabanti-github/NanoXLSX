615 private void AddNextCell(Cell cell, bool incremental, Style style)
616 {
617 // date and time styles are already defined by the passed cell object
618 if (style != null || (activeStyle != null && useActiveStyle))
619 {
620 if (cell.CellStyle == null && useActiveStyle)
621 {
622 cell.SetStyle(activeStyle);
623 }
624 else if (cell.CellStyle == null && style != null)
625 {
626 cell.SetStyle(style);
627 }
628 else if (cell.CellStyle != null && useActiveStyle)
629 {
630 Style mixedStyle = (Style)cell.CellStyle.Copy();
631 mixedStyle.Append(activeStyle);
632 cell.SetStyle(mixedStyle);
633 }
634 else if (cell.CellStyle != null && style != null)
635 {
636 Style mixedStyle = (Style)cell.CellStyle.Copy();
637 mixedStyle.Append(style);
638 cell.SetStyle(mixedStyle);
639 }
640 }
641 string address = cell.CellAddress;
642 if (cells.ContainsKey(address))
643 {
644 cells[address] = cell;
645 }
646 else
647 {
648 cells.Add(address, cell);
649 }
650 if (incremental)
651 {
652 if (CurrentCellDirection == CellDirection.ColumnToColumn)
653 {
654 currentColumnNumber++;
655 }
656 else if (CurrentCellDirection == CellDirection.RowToRow)
657 {
658 currentRowNumber++;
659 }
660 else
661 {
662 // disabled / no-op
663 }
664 }
665 else
666 {
667 if (CurrentCellDirection == CellDirection.ColumnToColumn)
668 {
669 currentColumnNumber = cell.ColumnNumber + 1;
670 currentRowNumber = cell.RowNumber;
671 }
672 else if (CurrentCellDirection == CellDirection.RowToRow)
673 {
674 currentColumnNumber = cell.ColumnNumber;
675 currentRowNumber = cell.RowNumber + 1;
676 }
677 else
678 {
679 // disabled / no-op
680 }
681 }

951 private void AddCellRangeInternal<T>(IReadOnlyList<T> values, Address startAddress, Address endAddress, Style style)
952 {
953 List<Address> addresses = Cell.GetCellRange(startAddress, endAddress) as List<Address>;
954 if (values.Count != addresses.Count)
955 {
956 throw new RangeException("The number of passed values (" + values.Count + ") differs from the number of cells within the range (" + addresses.Count + ")");
957 }
958 List<Cell> list = Cell.ConvertArray(values) as List<Cell>;
959 int len = values.Count;
960 for (int i = 0; i < len; i++)
961 {
962 list[i].RowNumber = addresses[i].Row;
963 list[i].ColumnNumber = addresses[i].Column;
964 AddNextCell(list[i], false, style);
965 }

1409 public void InsertRow(int rowNumber, int numberOfNewRows)
1410 {
1411 // All cells below the first row must receive a new address (row + count);
1412 var upperRow = this.GetRow(rowNumber);
1413
1414 // Identify all cells below the insertion point to adjust their addresses
1415 var cellsToChange = this.Cells.Where(c => c.Value.CellAddress2.Row > rowNumber).ToList();
1416
1417 // Make a copy of the cells to be moved and then delete the original cells;
1418 Dictionary<string, Cell> newCells = new Dictionary<string, Cell>();
1419 foreach (var cell in cellsToChange)
1420 {
1421 var row = cell.Value.CellAddress2.Row;
1422 var col = cell.Value.CellAddress2.Column;
1423 Address newAddress = new Address(col, row + numberOfNewRows);
1424
1425 Cell newCell = new Cell(cell.Value.Value, cell.Value.DataType, newAddress);
1426 if (cell.Value.CellStyle != null)
1427 {
1428 newCell.SetStyle(cell.Value.CellStyle); // Apply the style from the "old" cell.
1429 }
1430 newCells.Add(newAddress.GetAddress(), newCell);
1431
1432 // Delete the original cells since the key cannot be changed.
1433 this.Cells.Remove(cell.Key);
1434 }
1435
1436 // Fill the gap with new cells, using the same style as the first row.
1437 foreach (Cell cell in upperRow)
1438 {
1439 for (int i = 0; i < numberOfNewRows; i++)
1440 {
1441 Address newAddress = new Address(cell.CellAddress2.Column, cell.CellAddress2.Row + 1 + i);
1442 Cell newCell = new Cell(null, Cell.CellType.EMPTY, newAddress);
1443 if (cell.CellStyle != null)
1444 newCell.SetStyle(cell.CellStyle);
1445 this.Cells.Add(newAddress.GetAddress(), newCell);
1446 }
