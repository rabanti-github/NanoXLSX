260 public static FormatRange TryParseFormatNumber(int number, out FormatNumber formatNumber)
261 {
262
263 bool isDefined = System.Enum.IsDefined(typeof(FormatNumber), number);
264 if (isDefined)
265 {
266 formatNumber = (FormatNumber)number;
267 return FormatRange.DefinedFormat;
268 }
269 if (number < 0)
270 {
271 formatNumber = FormatNumber.none;
272 return FormatRange.Invalid;
273 }
274 else if (number > 0 && number < CustomFormatStartNumber)
275 {
276 formatNumber = FormatNumber.none;
277 return FormatRange.Undefined;
278 }
279 else
280 {
281 formatNumber = FormatNumber.custom;
282 return FormatRange.CustomFormat;
283 }
