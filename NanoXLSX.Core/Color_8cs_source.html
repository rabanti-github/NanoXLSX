302 public int CompareTo(object obj)
303 {
304 if (obj == null)
305 {
306 return 1;
307 }
308
309 if (!(obj is Color other))
310 {
311 throw new StyleException("The provided object to compare is not a Color");
312 }
313
314 // 1) Compare by color type first
315 int typeCompare = Type.CompareTo(other.Type);
316 if (typeCompare != 0)
317 {
318 return typeCompare;
319 }
320 // 2) Same type -> compare internal representation
321 switch (Type)
322 {
323 case ColorType.None:
324 return 0;
325 case ColorType.Auto: // Auto has no value
326 return 0;
327 case ColorType.Rgb:
328 return string.Compare(
329 RgbColor?.StringValue,
330 other.RgbColor?.StringValue,
331 StringComparison.OrdinalIgnoreCase);
332 case ColorType.Indexed:
333 return string.Compare(
334 IndexedColor?.StringValue,
335 other.IndexedColor?.StringValue,
336 StringComparison.Ordinal);
337 case ColorType.Theme:
338 {
339 int themeCompare = string.Compare(
340 ThemeColor?.StringValue,
341 other.ThemeColor?.StringValue,
342 StringComparison.Ordinal);
343
344 if (themeCompare != 0)
345 {
346 return themeCompare;
347 }
348 // Same theme index -> compare tint
349 return Nullable.Compare(Tint, other.Tint);
350 }
351 case ColorType.System:
352 return string.Compare(
353 SystemColor?.StringValue,
354 other.SystemColor?.StringValue,
355 StringComparison.OrdinalIgnoreCase);
356 default: // Defensive fallback
357 return string.Compare(
358 Value?.StringValue,
359 other.Value?.StringValue,
360 StringComparison.OrdinalIgnoreCase);
361 }
362 }
