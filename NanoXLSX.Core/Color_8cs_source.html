427 public int CompareTo(object obj)
428 {
429 if (obj == null)
430 {
431 return 1;
432 }
433
434 if (!(obj is Color other))
435 {
436 throw new StyleException("The provided object to compare is not a Color");
437 }
438
439 // 1) Compare by color type first
440 int typeCompare = Type.CompareTo(other.Type);
441 if (typeCompare != 0)
442 {
443 return typeCompare;
444 }
445
446 // 2) Same type -> compare internal representation
447 switch (Type)
448 {
449 case ColorType.None:
450 return 0;
451 case ColorType.Auto:
452 return 0;
453 case ColorType.Rgb:
454 return string.Compare(
455 RgbColor?.StringValue,
456 other.RgbColor?.StringValue,
457 StringComparison.OrdinalIgnoreCase);
458 case ColorType.Indexed:
459 // Numeric comparison of palette index
460 return IndexedColor.ColorValue.CompareTo(other.IndexedColor.ColorValue);
461 case ColorType.Theme:
462 {
463 // Numeric comparison of theme slot
464 int themeCompare = ThemeColor.ColorValue.CompareTo(other.ThemeColor.ColorValue);
465 if (themeCompare != 0)
466 {
467 return themeCompare;
468 }
469 // Same theme slot -> compare tint
470 return Nullable.Compare(Tint, other.Tint);
471 }
472 case ColorType.System:
473 // Enum-based comparison -> not string-based
474 return SystemColor.ColorValue.CompareTo(other.SystemColor.ColorValue);
475 default:
476 // Defensive fallback â> should normally never happen
477 return string.Compare(
478 Value?.StringValue,
479 other.Value?.StringValue,
480 StringComparison.OrdinalIgnoreCase);
481 }
482 }
