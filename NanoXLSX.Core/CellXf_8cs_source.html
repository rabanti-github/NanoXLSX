217 internal int CalculateInternalRotation()
218 {
219 if (textRotation < -90 || textRotation > 90)
220 {
221 throw new FormatException("The rotation value (" + ParserUtils.ToString(textRotation) + "°) is out of range. Range is form -90° to +90°");
222 }
223 if (textDirection == TextDirectionValue.vertical)
224 {
225 textRotation = 255;
226 return textRotation;
227 }
228 else
229 {
230 if (textRotation >= 0)
231 {
232 return textRotation;
233 }
234 else
235 {
236 return (90 - textRotation);
237 }
238 }
