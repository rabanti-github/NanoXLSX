240 public static int ParseBinaryBool(String rawValue)
241 {
242 if (string.IsNullOrEmpty(rawValue))
243 {
244 return 0;
245 }
246 int value;
247 if (TryParseInt(rawValue, out value))
248 {
249 if (value >= 1)
250 {
251 return 1;
252 }
253 else
254 {
255 return 0;
256 }
257 }
258 rawValue = rawValue.ToLower(InvariantCulture);
259 if (rawValue == "true")
260 {
261 return 1;
262 }
263 else
264 {
265 return 0;
266 }
