68 public void Execute()
69 {
70 this.styleReaderContainer = new StyleReaderContainer();
71 try
72 {
73 using (stream) // Close after processing
74 {
75 XmlDocument xr = new XmlDocument() { XmlResolver = null };
76 using (XmlReader reader = XmlReader.Create(stream, new XmlReaderSettings() { XmlResolver = null }))
77 {
78 xr.Load(reader);
79 foreach (XmlNode node in xr.DocumentElement.ChildNodes)
80 {
81 if (node.LocalName.Equals("numfmts", StringComparison.OrdinalIgnoreCase)) // Handles custom number formats
82 {
83 GetNumberFormats(node);
84 }
85 else if (node.LocalName.Equals("borders", StringComparison.OrdinalIgnoreCase)) // Handles borders
86 {
87 GetBorders(node);
88 }
89 else if (node.LocalName.Equals("fills", StringComparison.OrdinalIgnoreCase)) // Handles fills
90 {
91 GetFills(node);
92 }
93 else if (node.LocalName.Equals("fonts", StringComparison.OrdinalIgnoreCase)) // Handles fonts
94 {
95 GetFonts(node);
96 }
97 else if (node.LocalName.Equals("colors", StringComparison.OrdinalIgnoreCase)) // Handles MRU colors
98 {
99 GetColors(node);
100 }
101 // TODO: Implement other style components
102 }
103 foreach (XmlNode node in xr.DocumentElement.ChildNodes) // Redo for composition after all style parts are gathered; standard number formats
104 {
105 if (node.LocalName.Equals("cellxfs", StringComparison.OrdinalIgnoreCase))
106 {
107 GetCellXfs(node);
108 }
109 }
110 HandleMruColors();
111 RederPlugInHandler.HandleInlineQueuePlugins(ref stream, Workbook, PlugInUUID.StyleInlineReader);
112 }
113 }
114 Workbook.AuxiliaryData.SetData(PlugInUUID.StyleReader, PlugInUUID.StyleEntity, styleReaderContainer);
115 }
116 catch (Exception ex)
117 {
118 throw new IOException("The XML entry could not be read from the input stream. Please see the inner exception:", ex);
119 }
120 }

141 {

143 {

162 private void GetBorders(XmlNode node)
163 {
164 foreach (XmlNode border in node.ChildNodes)
165 {
166 Border borderStyle = new Border();
167 string diagonalDown = ReaderUtils.GetAttribute(border, "diagonalDown");
168 string diagonalUp = ReaderUtils.GetAttribute(border, "diagonalUp");
169 if (diagonalDown != null)
170 {
171 int value = ParserUtils.ParseBinaryBool(diagonalDown);
172 if (value == 1)
173 {
174 borderStyle.DiagonalDown = true;
175 }
176 }
177 if (diagonalUp != null)
178 {
179 int value = ParserUtils.ParseBinaryBool(diagonalUp);
180 if (value == 1)
181 {
182 borderStyle.DiagonalUp = true;
183 }
184 }
185 XmlNode innerNode = ReaderUtils.GetChildNode(border, "diagonal");
186 if (innerNode != null)
187 {
188 borderStyle.DiagonalStyle = ParseBorderStyle(innerNode);
189 borderStyle.DiagonalColor = GetColor(innerNode, Border.DefaultBorderColor);
190 }
191 innerNode = ReaderUtils.GetChildNode(border, "top");
192 if (innerNode != null)
193 {
194 borderStyle.TopStyle = ParseBorderStyle(innerNode);
195 borderStyle.TopColor = GetColor(innerNode, Border.DefaultBorderColor);
196 }
197 innerNode = ReaderUtils.GetChildNode(border, "bottom");
198 if (innerNode != null)
199 {
200 borderStyle.BottomStyle = ParseBorderStyle(innerNode);
201 borderStyle.BottomColor = GetColor(innerNode, Border.DefaultBorderColor);
202 }
203 innerNode = ReaderUtils.GetChildNode(border, "left");
204 if (innerNode != null)
205 {
206 borderStyle.LeftStyle = ParseBorderStyle(innerNode);
207 borderStyle.LeftColor = GetColor(innerNode, Border.DefaultBorderColor);
208 }
209 innerNode = ReaderUtils.GetChildNode(border, "right");
210 if (innerNode != null)
211 {
212 borderStyle.RightStyle = ParseBorderStyle(innerNode);
213 borderStyle.RightColor = GetColor(innerNode, Border.DefaultBorderColor);
214 }
215 borderStyle.InternalID = this.styleReaderContainer.GetNextBorderId();
216 this.styleReaderContainer.AddStyleComponent(borderStyle);
217 }

274 private Color ReadColorFromNode(XmlNode colorNode)
275 {
276 // Check for auto attribute
277 string autoAttr = ReaderUtils.GetAttribute(colorNode, "auto");
278 if (!string.IsNullOrEmpty(autoAttr) && ParserUtils.ParseBinaryBool(autoAttr) == 1)
279 {
280 return Color.CreateAuto();
281 }
282
283 // Check for rgb attribute
284 string rgbAttr = ReaderUtils.GetAttribute(colorNode, "rgb");
285 if (!string.IsNullOrEmpty(rgbAttr))
286 {
287 return Color.CreateRgb(rgbAttr);
288 }
289
290 // Check for indexed attribute
291 string indexedAttr = ReaderUtils.GetAttribute(colorNode, "indexed");
292 if (!string.IsNullOrEmpty(indexedAttr))
293 {
294 return Color.CreateIndexed(ParserUtils.ParseInt(indexedAttr));
295 }
296
297 // Check for theme attribute
298 string themeAttr = ReaderUtils.GetAttribute(colorNode, "theme");
299 if (!string.IsNullOrEmpty(themeAttr))
300 {
301 int themeIndex = ParserUtils.ParseInt(themeAttr);
302 // Check for optional tint attribute
303 string tintAttr = ReaderUtils.GetAttribute(colorNode, "tint");
304 double? tint = null;
305 if (!string.IsNullOrEmpty(tintAttr))
306 {
307 tint = ParserUtils.ParseDouble(tintAttr); // Or Convert.ToDouble with InvariantCulture
308 }
309 return Color.CreateTheme((Theme.ColorSchemeElement)themeIndex, tint);
310 }
311
312 // Check for system attribute (if supported)
313 string systemAttr = ReaderUtils.GetAttribute(colorNode, "system");
314 if (!string.IsNullOrEmpty(systemAttr))
315 {
316 SystemColor sysColor = new SystemColor(SystemColor.MapStringToValue(systemAttr));
317 return Color.CreateSystem(sysColor);
318 }
319
320 // No color defined
321 return Color.CreateNone();
322 }

565 private void GetCellXfs(XmlNode node)
566 {
567 foreach (XmlNode childNode in node.ChildNodes)
568 {
569 if (ReaderUtils.IsNode(childNode, "xf"))
570 {
571 CellXf cellXfStyle = new CellXf();
572 string attribute = ReaderUtils.GetAttribute(childNode, "applyAlignment");
573 if (attribute != null)
574 {
575 int value = ParserUtils.ParseBinaryBool(attribute);
576 cellXfStyle.ForceApplyAlignment = value == 1;
577 }
578 XmlNode alignmentNode = ReaderUtils.GetChildNode(childNode, "alignment");
579 if (alignmentNode != null)
580 {
581 attribute = ReaderUtils.GetAttribute(alignmentNode, "shrinkToFit");
582 if (attribute != null)
583 {
584 int value = ParserUtils.ParseBinaryBool(attribute);
585 if (value == 1)
586 {
587 cellXfStyle.Alignment = TextBreakValue.ShrinkToFit;
588 }
589 }
590 attribute = ReaderUtils.GetAttribute(alignmentNode, "wrapText");
591 if (attribute != null && attribute == "1")
592 {
593 cellXfStyle.Alignment = TextBreakValue.WrapText;
594 }
595 attribute = ReaderUtils.GetAttribute(alignmentNode, "horizontal", string.Empty);
596 cellXfStyle.HorizontalAlign = CellXf.GetHorizontalAlignEnum(attribute);
597 attribute = ReaderUtils.GetAttribute(alignmentNode, "vertical", string.Empty);
598 cellXfStyle.VerticalAlign = CellXf.GetVerticalAlignEnum(attribute);
599 attribute = ReaderUtils.GetAttribute(alignmentNode, "indent");
600 if (attribute != null)
601 {
602 cellXfStyle.Indent = ParserUtils.ParseInt(attribute);
603 }
604 attribute = ReaderUtils.GetAttribute(alignmentNode, "textRotation");
605 if (attribute != null)
606 {
607 int rotation = ParserUtils.ParseInt(attribute);
608 cellXfStyle.TextRotation = rotation > 90 ? 90 - rotation : rotation;
609 }
610 }
611 XmlNode protectionNode = ReaderUtils.GetChildNode(childNode, "protection");
612 if (protectionNode != null)
613 {
614 attribute = ReaderUtils.GetAttribute(protectionNode, "hidden");
615 if (attribute != null && attribute == "1")
616 {
617 cellXfStyle.Hidden = true;
618 }
619 attribute = ReaderUtils.GetAttribute(protectionNode, "locked");
620 if (attribute != null && attribute == "0")
621 {
622 cellXfStyle.Locked = false;
623 }
624 // else - NoOp - No need to set locked value, since true by default
625 }
626
627 cellXfStyle.InternalID = this.styleReaderContainer.GetNextCellXFId();
628 this.styleReaderContainer.AddStyleComponent(cellXfStyle);
629
630 Style style = new Style();
631 int id;
632 bool hasId;
633
634 hasId = ParserUtils.TryParseInt(ReaderUtils.GetAttribute(childNode, "numFmtId"), out id);
635 NumberFormat format = this.styleReaderContainer.GetNumberFormat(id);
636 if (!hasId || format == null)
637 {
638 FormatNumber formatNumber;
639 NumberFormat.TryParseFormatNumber(id, out formatNumber); // Validity is neglected here to prevent unhandled crashes. If invalid, the format will be declared as 'none'.
640 // Invalid values should not occur at all (malformed Excel files).
641 // Undefined values may occur if the file was saved by an Excel version that has implemented yet unknown format numbers (undefined in NanoXLSX)
642 format = new NumberFormat
643 {
644 Number = formatNumber,
645 InternalID = id
646 };
647 this.styleReaderContainer.AddStyleComponent(format);
648 }
649 hasId = ParserUtils.TryParseInt(ReaderUtils.GetAttribute(childNode, "borderId"), out id);
650 Border border = this.styleReaderContainer.GetBorder(id);
651 if (!hasId || border == null)
652 {
653 border = new Border
654 {
655 InternalID = this.styleReaderContainer.GetNextBorderId()
656 };
657 }
658 hasId = ParserUtils.TryParseInt(ReaderUtils.GetAttribute(childNode, "fillId"), out id);
659 Fill fill = this.styleReaderContainer.GetFill(id);
660 if (!hasId || fill == null)
661 {
662 fill = new Fill
663 {
664 InternalID = this.styleReaderContainer.GetNextFillId()
665 };
666 }
667 hasId = ParserUtils.TryParseInt(ReaderUtils.GetAttribute(childNode, "fontId"), out id);
668 Font font = this.styleReaderContainer.GetFont(id);
669 if (!hasId || font == null)
670 {
671 font = new Font
672 {
673 InternalID = this.styleReaderContainer.GetNextFontId()
674 };
675 }
676
677 // TODO: Implement other style information
678 style.CurrentNumberFormat = format;
679 style.CurrentBorder = border;
680 style.CurrentFill = fill;
681 style.CurrentFont = font;
682 style.CurrentCellXf = cellXfStyle;
683 style.InternalID = this.styleReaderContainer.GetNextStyleId();
684
685 this.styleReaderContainer.AddStyleComponent(style);
686 }
687 }
